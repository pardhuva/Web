        if(root == null || subroot ==null) return false;
        
        if(root.val == subroot.val){

            if(issubtree(root.left,subroot.left) && issubtree(root.right,subroot.right)){

    public static Node createnode(int i , int[] arr){

            return null;
        }

    }
    public static void levelorder(Node root){
        Queue<Node> q = new LinkedList<>();
        
        if(root == null){
            return;
        }
        q.add(root);
        q.add(null);
        while(!q.isEmpty()){
            
            Node node = q.remove();
        
            if(node == null){
                q.add(null);
                System.out.println();
            }else{
                System.out.print(node.val);
                
                if(node.left!= null){
                    q.add(node.left);
                }
                if(node.right!=null){
                    q.add(node.right);
                }
            }
        }
        
    }
    public static void main(String[] args) {
        int[] arr = {3,2,4,5,7,6,8,9};
        
        Node root = createnode(0,arr);
        levelorder(root);
    }
}
